384 # Create anonymous composite w = pywrap(o) wrapping the object o
385 # and providing access to o's members (converted to PyAny) as w.member.
386
387 # we skip wrapping Julia reserved words (which cannot be type members)
388 const reserved = Set{String}(["while", "if", "for", "try", "return", "break", "continue", "function", "macro", "quote", "let", "local", "global", "const", "abstract", "typealias", "type", "bitstype", "immutable", "ccall", "do", "module", "baremodule", "using", "import", "export", "importall", "pymember", "false", "true", "Tuple"])

390 function _pywrap(o::PyObject, mname::Symbol)
391 members = convert(Vector{Tuple{AbstractString,PyObject}},
392 pycall(inspect."getmembers", PyObject, o))
393 filter!(m -> !(m[1] in reserved), members)
394 m = Module(mname, false)
395 consts = [Expr(:const, Expr(:(=), Symbol(x[1]), convert(PyAny, x[2]))) for x in members]
396 exports = try
397 convert(Vector{Symbol}, o."__all__")
398 catch
399 [Symbol(x[1]) for x in filter(x -> x[1][1] != '_', members)]
400 end
401 Core.eval(m, Expr(:toplevel, consts..., :(pymember(s) = $(getindex)($(o), s)),
402 Expr(:export, exports...)))
403 m
404 end

617 @pywith EXPR[::TYPE1] [as VAR[::TYPE2]] begin
618 BLOCK
619 end
620
621 TYPE1/TYPE2 can optionally be used to override automatic conversion to Julia
622 types for both the context manager and variable in cases where this is not
623 desired.
624
625 """
626 macro pywith(EXPR,as,VAR,BLOCK)
627 if isexpr(VAR,:(::))
628 VAR,TYPE = VAR.args
629 else
630 TYPE = PyAny
631 end
632 if !((as==:as) && isa(VAR,Symbol))
633 throw(ArgumentError("usage: @pywith EXPR[::TYPE1] [as VAR[::TYPE2]] BLOCK."))
634 end
635 _pywith(EXPR,VAR,TYPE,BLOCK)
636 end
637 macro pywith(EXPR,BLOCK)
638 _pywith(EXPR,nothing,PyObject,BLOCK)
639 end
640
641
642 function _pywith(EXPR,VAR,TYPE,BLOCK)
643 EXPR_str = string(EXPR)
644 quote
645 mgr = $(isexpr(EXPR,:(::)) ? :(@pycall $(esc(EXPR))) : esc(EXPR))
646 if !isa(mgr,PyObject)
647 if $(isexpr(EXPR,:call))
648 throw(ArgumentError("@pywith: `$($EXPR_str)` did not return a PyObject. If this is a call to a Python function, try `$($EXPR_str)::PyObject` to turn off automatic conversion."))
649 else
650 throw(ArgumentError("@pywith: `$($EXPR_str)` should be a PyObject."))
651 end
652 end
653 mgrT = pytypeof(mgr)
654 exit = mgrT."__exit__"
655 value = @pycall mgrT."__enter__"(mgr)::$(esc(TYPE))
656 exc = true
657 try
658 try
659 $(VAR==nothing ? :() : :($(esc(VAR)) = value))
660 $(esc(BLOCK))
661 catch err
662 exc = false
663 if !(@pycall exit(mgr, pyimport(:sys).exc_info()...)::Bool)
664 throw(err)
665 end
666 end
667 finally
668 if exc
669 exit(mgr, nothing, nothing, nothing)
670 end
671 end
672 nothing
673 end
674 end

679 """
680 anaconda_conda()
681
682 Return the path of the `conda` program if PyCall is configured to use
683 an Anaconda install (but *not* the Conda.jl Python), and the empty
684 string otherwise.
685 """
686 function anaconda_conda()
687 # Anaconda Python seems to always include "Anaconda" in the version string.
688 if conda || !occursin("conda", unsafe_string(ccall(@pysym(:Py_GetVersion), Ptr{UInt8}, ())))
689 return ""
690 end
691 aconda = joinpath(dirname(pyprogramname), "conda")
692 return isfile(aconda) ? aconda : ""

695 """
696 pyimport_conda(modulename, condapkg, [channel])
697
698 Returns the result of `pyimport(modulename)` if possible. If the module
699 is not found, and PyCall is configured to use the Conda Python distro
700 (via the Julia Conda package), then automatically install `condapkg`
701 via `Conda.add(condapkg)` and then re-try the `pyimport`. Other
702 Anaconda-based Python installations are also supported as long as
703 their `conda` program is functioning.
704
705 If PyCall is not using Conda and the `pyimport` fails, throws
706 an exception with an error message telling the user how to configure
707 PyCall to use Conda for automated installation of the module.
708
709 The third argument, `channel` is an optional Anaconda "channel" to use
710 for installing the package; this is useful for packages that are not
711 included in the default Anaconda package listing.
712 """
713 function pyimport_conda(modulename::AbstractString, condapkg::AbstractString,
714 channel::AbstractString="")
715 try
716 pyimport(modulename)
717 catch e
718 if conda
719 @info "Installing $modulename via the Conda $condapkg package..."
720 isempty(channel) || Conda.add_channel(channel)
721 Conda.add(condapkg)
722 pyimport(modulename)
723 else
724 aconda = anaconda_conda()
725 if isempty(aconda)
726 # Not in conda environment; show the error from `pyimport`.
727 rethrow()
728 end
729
730 options = ""
731 if !isempty(channel)
732 options = " --channel $channel"
733 end
734
735 error("""
736 Failed to import required Python module $modulename.
737
738 For automated $modulename installation, try configuring PyCall to use the Conda.jl package's Python "Miniconda" distribution within Julia. Relaunch Julia and run:
739 ENV["PYTHON"]=""
740 Pkg.build("PyCall")
741 before trying again.
742
743 Note that Conda.jl can use custom conda environment. Read more about `CONDA_JL_HOME` configuration in Conda.jl documentation:
744 https://github.com/JuliaPy/Conda.jl
745
746 To install $condapkg using $aconda, you can run the following command from your system shell:
747 $aconda install$options $condapkg
748
749 The pyimport exception was: $e
750 """)
751 end
752 end

895 # support IPython _repr_foo functions for MIME output of PyObjects

897 for (mime, method) in ((MIME"text/html", "_repr_html_"),
898 (MIME"text/markdown", "_repr_markdown_"),
899 (MIME"text/json", "_repr_json_"),
900 (MIME"application/javascript", "_repr_javascript_"),

902 (MIME"image/jpeg", "_repr_jpeg_"),
903 (MIME"image/png", "_repr_png_"),
904 (MIME"image/svg+xml", "_repr_svg_"),
905 (MIME"text/latex", "_repr_latex_"))
906 T = istextmime(mime()) ? AbstractString : Vector{UInt8}
907 @eval begin
908 function show(io::IO, mime::$mime, o::PyObject)
909 m = get_real_method(o, $method)
910 if m !== nothing
911 r = pycall(m, PyObject)
912 !(r ≛ pynothing[]) && return write(io, convert($T, r))
913 end
914 throw(MethodError(show, (io, mime, o)))
915 end

918 m === nothing && return false

920 end
921 end

924 """
925 get_real_method(obj::PyObject, name::String) -> method::Union{PyObject,Nothing}
926
927 This is a port of `IPython.utils.dir2.get_real_method`:
928 https://github.com/ipython/ipython/blob/7.9.0/IPython/utils/dir2.py
929
930 For Python 2-era
931 https://github.com/ipython/ipython/blob/5.9.0/IPython/utils/dir2.py
932 """
933 function get_real_method(obj, name)
934 ispynull(obj) && return nothing
935 @static if pyversion < v"3"
936 pyisinstance(obj, @pyglobalobj :PyType_Type) && return nothing
937 end
938
939 canary = try
940 trygetproperty(obj, "_ipython_canary_method_should_not_exist_", nothing)
941 catch
942 nothing
943 end
944
945 # It claimed to have an attribute it should never have
946 canary === nothing || return nothing
947
948 m = try
949 trygetproperty(obj, name, nothing)
950 catch
951 nothing
952 end
953 m === nothing && return nothing
954
955 if (
956 pyisinstance(obj, @pyglobalobj :PyType_Type) &&
957 !pyisinstance(m, @pyglobalobj :PyMethod_Type)
958 )
959 return nothing
960 end
961
962 ccall((@pysym :PyCallable_Check), Cint, (PyPtr,), m) == 1 && return m
963
964 return nothing
965 end
