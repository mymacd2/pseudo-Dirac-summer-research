6 # Linking NumPy's C API from Julia requires some serious hackery,
7 # because NumPy does not export its symbols in the usual way for
8 # shared libraries. Instead, it provides a Python variable
9 # numpy.core.multiarray._ARRAY_API that points to a lookup table of
10 # pointers to the API functions and global variables. Moreover, the
11 # meaning of this table, along with a lot of other important
12 # constants, is defined in a C header file which changes between NumPy
13 # versions, so we need to do some regex parsing of this header file in
14 # order to extract the necessary information to call NumPy. Ugly and
15 # mildly insane, but I don't see much alternative (at least to do this
16 # purely in Julia).
17 #
18 # The result of npy_api_initialize, below, is to produce the following
19 # tables of API pointers:
20 const npy_api = Dict{Symbol, Ptr{Cvoid}}()

92 # Parse __multiarray_api.h to obtain length and meaning of PyArray_API
93 try
94 hdrfile = open(joinpath(inc, "numpy", "__multiarray_api.h"))
95 hdr = read(hdrfile, String)

99 end
100 hdr = replace(hdr, r"\\\s*\n"=>" "); # rm backslashed newlines
101 r = r"^#define\s+([A-Za-z]\w*)\s+\(.*\bPyArray_API\s*\[\s*([0-9]+)\s*\]\s*\)\s*$"m # regex to match #define PyFoo (... PyArray_API[nnn])
102 PyArray_API_length = 0
103 for m in eachmatch(r, hdr) # search for max index into PyArray_API
104 PyArray_API_length = max(PyArray_API_length, parse(Int, m.captures[2])+1)
105 end
106 API = unsafe_wrap(Array, PyArray_API, (PyArray_API_length,))
107 for m in eachmatch(r, hdr) # build npy_api table
108 npy_api[Symbol(m.captures[1])] = API[parse(Int, m.captures[2])+1]
109 end
