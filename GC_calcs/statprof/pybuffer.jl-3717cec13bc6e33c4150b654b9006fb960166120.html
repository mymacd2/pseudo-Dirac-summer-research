152 # recursive function to write buffer dimension by dimension, starting at
153 # dimension d with the given pointer offset (in bytes).
154 function writedims(io::IO, b::PyBuffer, offset, d)
155 n = 0
156 s = stride(b, d)
157 if d < b.buf.ndim
158 for i = 1:size(b,d)
159 n += writedims(io, b, offset, d+1)
160 offset += s
161 end
162 else
163 @assert d == b.buf.ndim
164 p = convert(Ptr{UInt8}, pointer(b)) + offset
165 for i = 1:size(b,d)
166 # would be nicer not to write this one byte at a time,
167 # but the alternative seems to be to create an Array
168 # object on each loop iteration.
169 for j = 1:b.buf.itemsize
170 n += write(io, unsafe_load(p))
171 p += 1
172 end
173 p += s
174 end
175 end
176 return n
177 end
