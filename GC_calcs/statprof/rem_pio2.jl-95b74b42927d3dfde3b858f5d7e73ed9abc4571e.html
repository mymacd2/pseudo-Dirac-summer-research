98 """
99 fromfraction(f::Int128)
100 Compute a tuple of values `(z1,z2)` such that
101 ``z1 + z2 == f / 2^128``
102 and the significand of `z1` has 27 trailing zeros.
103 """
104 function fromfraction(f::Int128)
105 if f == 0
106 return (0.0,0.0)
107 end
108
109 # 1. get leading term truncated to 26 bits
110 s = ((f < 0) % UInt64) << 63 # sign bit
111 x = abs(f) % UInt128 # magnitude
112 n1 = 128-leading_zeros(x) # ndigits0z(x,2)
113 m1 = ((x >> (n1-26)) % UInt64) << 27
114 d1 = ((n1-128+1021) % UInt64) << 52
115 z1 = reinterpret(Float64, s | (d1 + m1))
116
117 # 2. compute remaining term
118 x2 = (x - (UInt128(m1) << (n1-53)))
119 if x2 == 0
120 return (z1, 0.0)
121 end
122 n2 = 128-leading_zeros(x2)
123 m2 = (x2 >> (n2-53)) % UInt64
124 d2 = ((n2-128+1021) % UInt64) << 52
125 z2 = reinterpret(Float64, s | (d2 + m2))
126 return (z1,z2)
127 end

130 # because the effect analysis currently can't prove it in the presence of `@inbounds` or
131 # `:boundscheck`, but still the accesses to `INV_2PI` are really safe here
132 Base.@assume_effects :consistent function paynehanek(x::Float64)
133 # 1. Convert to form
134 #
135 # x = X * 2^k,
136 #
137 # where 2^(n-1) <= X < 2^n is an n-bit integer (n = 53, k = exponent(x)-52 )
138
139 # Computations are integer based, so reinterpret x as UInt64
140 u = reinterpret(UInt64, x)
141 # Strip x of exponent bits and replace with ^1
142 X = (u & significand_mask(Float64)) | (one(UInt64) << significand_bits(Float64))
143 # Get k from formula above
144 # k = exponent(x)-52
145 raw_exponent = ((u & exponent_mask(Float64)) >> significand_bits(Float64)) % Int
146 k = raw_exponent - exponent_bias(Float64) - significand_bits(Float64)
147
148 # 2. Let α = 1/2π, then:
149 #
150 # α*x mod 1 ≡ [(α*2^k mod 1)*X] mod 1
151 #
152 # so we can ignore the first k bits of α. Extract the next 3 64-bit parts of α.
153 #
154 # i.e. equivalent to
155 # setprecision(BigFloat,4096)
156 # α = 1/(2*big(pi))
157 # A = mod(ldexp(α,k), 1)
158 # z1 = ldexp(A,64)
159 # a1 = trunc(UInt64, z1)
160 # z2 = ldexp(z1-a1, 64)
161 # a2 = trunc(UInt64, z2)
162 # z3 = ldexp(z2-a2, 64)
163 # a3 = trunc(UInt64, z3)
164
165 # This is equivalent to
166 # idx, shift = divrem(k, 64)
167 # but divrem is slower.
168 idx = k >> 6
169
170 shift = k - (idx << 6)
171 if shift == 0
172 @inbounds a1 = INV_2PI[idx+1]
173 @inbounds a2 = INV_2PI[idx+2]
174 @inbounds a3 = INV_2PI[idx+3]
175 else
176 # use shifts to extract the relevant 64 bit window
177 @inbounds a1 = (idx < 0 ? zero(UInt64) : INV_2PI[idx+1] << shift) | (INV_2PI[idx+2] >> (64 - shift))
178 @inbounds a2 = (INV_2PI[idx+2] << shift) | (INV_2PI[idx+3] >> (64 - shift))
179 @inbounds a3 = (INV_2PI[idx+3] << shift) | (INV_2PI[idx+4] >> (64 - shift))
180 end
181
182 # 3. Perform the multiplication:
183 #
184 # X. 0 0 0
185 # × 0. a1 a2 a3
186 # ==============
187 # _. w w _
188 #
189 # (i.e. ignoring integer and lowest bit parts of result)
190
191 w1 = UInt128(X*a1) << 64 # overflow becomes integer
192 w2 = widemul(X,a2)
193 w3 = widemul(X,a3) >> 64
194 w = w1 + w2 + w3 # quotient fraction after division by 2π
195
196 # adjust for sign of x
197 w = flipsign(w,x)
198
199 # 4. convert to quadrant, quotient fraction after division by π/2:
200 q = (((w>>125)%Int +1)>>1) # nearest quadrant
201 f = (w<<2) % Int128 # fraction part of quotient after division by π/2, taking values on [-0.5,0.5)
202
203 # 5. convert quotient fraction to split precision Float64
204 z_hi,z_lo = fromfraction(f)
205
206 # 6. multiply by π/2
207 pio2 = 1.5707963267948966
208 pio2_hi = 1.5707963407039642
209 pio2_lo = -1.3909067614167116e-8
210 y_hi = (z_hi+z_lo)*pio2
211 y_lo = (((z_hi*pio2_hi - y_hi) + z_hi*pio2_lo) + z_lo*pio2_hi) + z_lo*pio2_lo
212 return q, DoubleFloat64(y_hi, y_lo)
213 end
