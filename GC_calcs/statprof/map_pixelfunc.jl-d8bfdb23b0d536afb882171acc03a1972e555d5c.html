256 udgrade(input_map::HealpixMap{T,O,AA}, output_nside; kw...) where {T,O,AA} -> HealpixMap{T,O,AA}
257
258 Upgrades or downgrades a map to a target nside. Always makes a copy. This is very fast
259 for nested orderings, but slow for ring because one needs to transform to nested ordering
260 first.
261
262 # Arguments:
263 - `input_map::HealpixMap{T,O,AA}`: the map to upgrade/downgrade
264 - `output_nside`: desired nside
265
266 # Keywords:
267 - `threshold=abs(1e-6UNSEEN)`: absolute tolerance for identifying a bad pixel vs UNSEEN
268 - `pess=false`: if false, estimate pixels from remaining good pixels when downgrading.
269 if true, the entire downgraded pixel is set to UNSEEN.
270
271 # Returns:
272 - `HealpixMap{T,O,AA}`: upgraded/downgraded map in the same ordering as the input
273
274 # Examples
275 ```julia-repl
276 julia> A = HealpixMap{Float64, NestedOrder}(ones(nside2npix(4)))
277 192-element HealpixMap{Float64, RingOrder, Vector{Float64}}:
278 1.0
279 ⋮
280 1.0
281
282 julia> Healpix.udgrade(A, 2)
283 48-element HealpixMap{Float64, NestedOrder, Vector{Float64}}:
284 1.0
285 ⋮
286 1.0
287 ```
288 """
289 function udgrade(map_in::HealpixMap{T,O,AA}, nside_out;
290 threshold=abs(1e-6UNSEEN), pess=false) where {T,O<:NestedOrder,AA}
291 nside_in = map_in.resolution.nside
292 npix_out = nside2npix(nside_out)
293 map_out = HealpixMap{T,O,AA}(nside_out)
294
295 if nside_in == nside_out
296 map_out.pixels .= map_in.pixels
297 elseif nside_out < nside_in # degrade. loop over input and add them up
298 npratio = nside2npix(nside_in) ÷ nside2npix(nside_out)
299 for id = 0:(npix_out-1)
300 nobs = 0
301 total = 0.0
302 for ip = 0:(npratio-1)
303 value = map_in[id*npratio + ip + 1]
304 if (abs(value - UNSEEN) > threshold)
305 nobs = nobs + 1
306 total = total + value
307 end
308 end
309 map_out[id+1] = UNSEEN
310 if pess
311 if nobs == npratio
312 map_out[id+1] = total/nobs
313 end
314 else
315 if nobs > 0
316 map_out[id+1] = total/nobs
317 end
318 end
319 end
320 else # we are upgrading. loop over output map pixels, and set them to parent pixel
321 npratio = nside2npix(nside_out) ÷ nside2npix(nside_in)
322 for iu = 0:(npix_out - 1)
323 ip = iu ÷ npratio
324 map_out[iu+1] = map_in[ip+1]
325 end
326 end
327
328 return map_out
329 end

332 threshold=abs(1e-6UNSEEN), pess=false) where {T,O<:RingOrder,AA}
333 map_nest = ring2nest(map_in)
334 map_out = udgrade(map_nest, nside_out; threshold=threshold, pess=pess)
