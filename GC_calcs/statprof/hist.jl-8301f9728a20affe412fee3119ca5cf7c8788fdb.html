514 normalize(h::Histogram{T,N}; mode::Symbol=:pdf) where {T,N}
515
516 Normalize the histogram `h`.
517
518 Valid values for `mode` are:
519
520 * `:pdf`: Normalize by sum of weights and bin sizes. Resulting histogram
521 has norm 1 and represents a PDF.
522 * `:density`: Normalize by bin sizes only. Resulting histogram represents
523 count density of input and does not have norm 1. Will not modify the
524 histogram if it already represents a density (`h.isdensity == 1`).

526 represents the fraction of probability mass for each bin and does not have

566 """
567 merge!(target::Histogram, others::Histogram...)
568
569 Update histogram `target` by merging it with the histograms `others`. See
570 `merge(histogram::Histogram, others::Histogram...)` for details.
571 """
572 function Base.merge!(target::Histogram, others::Histogram...)
573 for h in others
574 target.edges != h.edges && throw(ArgumentError("can't merge histograms with different binning"))
575 size(target.weights) != size(h.weights) && throw(ArgumentError("can't merge histograms with different dimensions"))
576 target.closed != h.closed && throw(ArgumentError("can't merge histograms with different closed left/right settings"))
577 target.isdensity != h.isdensity && throw(ArgumentError("can't merge histograms with different isdensity settings"))
578 end
579 for h in others
580 target.weights .+= h.weights
581 end
582 target
583 end

586 """
587 merge(h::Histogram, others::Histogram...)
588
589 Construct a new histogram by merging `h` with `others`. All histograms must
590 have the same binning, shape of weights and properties (`closed` and
591 `isdensity`). The weights of all histograms are summed up for each bin, the
592 weights of the resulting histogram will have the same type as those of `h`.
593 """
594 Base.merge(h::Histogram, others::Histogram...) = merge!(zero(h), h, others...)
