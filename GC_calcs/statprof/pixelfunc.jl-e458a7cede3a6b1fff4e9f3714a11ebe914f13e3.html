630 """
631 Create an array of the colatitude in radians (theta) of each ring index in `rings` for a map with resolution `res`.
632
633 If no `rings` array is passed, the computation is performed on all the rings deducted from `res`.
634
635 If an integer `ring` is passed, a single Float value of the colatitude is returned.
636 """
637 function ring2theta(rings::Vector{I}, res::Resolution) where {I<:Integer}
638 theta = Vector{Float64}(undef, length(rings))
639 ringinfo = RingInfo(0, 0, 0, 0, 0)
640 @inbounds for i in 1:length(rings)
641 getringinfo!(res, rings[i], ringinfo; full=true)
642 theta[i] = ringinfo.colatitude_rad
643 end
644 theta
645 end
646 ring2theta(res::Resolution) = ring2theta(Vector{Int}(1:res.nsideTimesFour-1), res)

648 ring2theta(ring::Integer, res::Resolution) = getringinfo(res, ring; full=true).colatitude_rad
