359 """
360 logdiffcdf(d::UnivariateDistribution, x::Real, y::Real)
361
362 The natural logarithm of the difference between the cumulative density function at `x` and `y`, i.e. `log(cdf(x) - cdf(y))`.
363 """
364 function logdiffcdf(d::UnivariateDistribution, x::Real, y::Real)
365 # Promote to ensure that we don't compute logcdf in low precision and then promote
366 _x, _y = promote(x, y)
367 _x < _y && throw(ArgumentError("requires x >= y."))
368 u = logcdf(d, _x)
369 v = logcdf(d, _y)
370 return u + log1mexp(v - u)
371 end

415 function _pdf_fill_outside!(r::AbstractArray, d::DiscreteUnivariateDistribution, X::UnitRange)
416 vl = vfirst = first(X)
417 vr = vlast = last(X)
418 n = vlast - vfirst + 1
419 if islowerbounded(d)
420 lb = minimum(d)
421 if vl < lb
422 vl = lb
423 end
424 end
425 if isupperbounded(d)
426 ub = maximum(d)
427 if vr > ub
428 vr = ub
429 end
430 end
431
432 # fill left part
433 if vl > vfirst
434 for i = 1:(vl - vfirst)
435 r[i] = 0.0
436 end
437 end
438
439 # fill central part: with non-zero pdf
440 fm1 = vfirst - 1
441 for v = vl:vr
442 r[v - fm1] = pdf(d, v)
443 end
444
445 # fill right part
446 if vr < vlast
447 for i = (vr-vfirst+2):n
448 r[i] = 0.0
449 end
450 end
451 return vl, vr, vfirst, vlast
452 end

549 # implementation of the cdf for distributions whose support is a unitrange of integers
550 # note: incorrect for discrete distributions whose support includes non-integer numbers
551 function integerunitrange_cdf(d::DiscreteUnivariateDistribution, x::Integer)
552 minimum_d, maximum_d = extrema(d)
553 isfinite(minimum_d) || isfinite(maximum_d) || error("support is unbounded")
554
555 result = if isfinite(minimum_d) && !(isfinite(maximum_d) && x >= div(minimum_d + maximum_d, 2))
556 c = sum(Base.Fix1(pdf, d), minimum_d:(max(x, minimum_d)))
557 x < minimum_d ? zero(c) : c
558 else
559 c = 1 - sum(Base.Fix1(pdf, d), (min(x + 1, maximum_d)):maximum_d)
560 x >= maximum_d ? one(c) : c
561 end
562
563 return result
564 end
