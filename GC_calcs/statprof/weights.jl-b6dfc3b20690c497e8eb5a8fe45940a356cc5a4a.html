215 eweights(n::Integer, λ::Real; scale=false)
216
217 Construct a [`Weights`](@ref) vector which assigns exponentially decreasing weights to past
218 observations (larger integer values `i` in `t`).
219 The integer value `n` represents the number of past observations to consider.
220 `n` defaults to `maximum(t) - minimum(t) + 1` if only `t` is passed in
221 and the elements are integers, and to `length(r)` if a superset range `r` is also passed in.
222 If `n` is explicitly passed instead of `t`, `t` defaults to `1:n`.
223
224 If `scale` is `true` then for each element `i` in `t` the weight value is computed as:
225
226 ``(1 - λ)^{n - i}``
227
228 If `scale` is `false` then each value is computed as:
229
230 ``λ (1 - λ)^{1 - i}``
231
232 # Arguments
233
234 - `t::AbstractVector`: temporal indices or timestamps
235 - `r::StepRange`: a larger range to use when constructing weights from a subset of timestamps
236 - `n::Integer`: the number of past events to consider
237 - `λ::Real`: a smoothing factor or rate parameter such that ``0 < λ ≤ 1``.
238 As this value approaches 0, the resulting weights will be almost equal,
239 while values closer to 1 will put greater weight on the tail elements of the vector.
240
241 # Keyword arguments
242
243 - `scale::Bool`: Return the weights scaled to between 0 and 1 (default: false)
244
245 # Examples
246 ```julia-repl
247 julia> eweights(1:10, 0.3; scale=true)

404 ## wsum along dimension
405 #
406 # Brief explanation of the algorithm:
407 # ------------------------------------
408 #
409 # 1. _wsum! provides the core implementation, which assumes that
410 # the dimensions of all input arguments are consistent, and no
411 # dimension checking is performed therein.
412 #
413 # wsum and wsum! perform argument checking and call _wsum!
414 # internally.
415 #
416 # 2. _wsum! adopt a Cartesian based implementation for general
417 # sub types of AbstractArray. Particularly, a faster routine
418 # that keeps a local accumulator will be used when dim = 1.
419 #
420 # The internal function that implements this is _wsum_general!
421 #
422 # 3. _wsum! is specialized for following cases:
423 # (a) A is a vector: we invoke the vector version wsum above.
424 # The internal function that implements this is _wsum1!
425 #
426 # (b) A is a dense matrix with eltype <: BlasReal: we call gemv!
427 # The internal function that implements this is _wsum2_blas!
428 #
429 # (c) A is a contiguous array with eltype <: BlasReal:
430 # dim == 1: treat A like a matrix of size (d1, d2 x ... x dN)
431 # dim == N: treat A like a matrix of size (d1 x ... x d(N-1), dN)
432 # otherwise: decompose A into multiple pages, and apply _wsum2!
433 # for each
434 #
435 # (d) A is a general dense array with eltype <: BlasReal:
436 # dim <= 2: delegate to (a) and (b)
437 # otherwise, decompose A into multiple pages
438
439 function _wsum1!(R::AbstractArray, A::AbstractVector, w::AbstractVector, init::Bool)
440 r = wsum(A, w)
441 if init
442 R[1] = r
443 else
444 R[1] += r
445 end
446 return R
447 end

678 ##### Weighted quantile #####
679
680 """
681 quantile(v, w::AbstractWeights, p)
682
683 Compute the weighted quantiles of a vector `v` at a specified set of probability
684 values `p`, using weights given by a weight vector `w` (of type `AbstractWeights`).
685 Weights must not be negative. The weights and data vectors must have the same length.
686 `NaN` is returned if `x` contains any `NaN` values. An error is raised if `w` contains
687 any `NaN` values.
688
689 With [`FrequencyWeights`](@ref), the function returns the same result as
690 `quantile` for a vector with repeated values. Weights must be integers.
691
692 With non `FrequencyWeights`, denote ``N`` the length of the vector, ``w`` the vector of weights,
693 ``h = p (\\sum_{i<= N} w_i - w_1) + w_1`` the cumulative weight corresponding to the
694 probability ``p`` and ``S_k = \\sum_{i<=k} w_i`` the cumulative weight for each
695 observation, define ``v_{k+1}`` the smallest element of `v` such that ``S_{k+1}``
696 is strictly superior to ``h``. The weighted ``p`` quantile is given by ``v_k + \\gamma (v_{k+1} - v_k)``
697 with ``\\gamma = (h - S_k)/(S_{k+1} - S_k)``. In particular, when all weights are equal,
698 the function returns the same result as the unweighted `quantile`.
699 """
700 function quantile(v::RealVector{V}, w::AbstractWeights{W}, p::RealVector) where {V,W<:Real}
701 # checks
702 isempty(v) && throw(ArgumentError("quantile of an empty array is undefined"))
703 isempty(p) && throw(ArgumentError("empty quantile array"))
704 all(x -> 0 <= x <= 1, p) || throw(ArgumentError("input probability out of [0,1] range"))
705
706 w.sum == 0 && throw(ArgumentError("weight vector cannot sum to zero"))
707 length(v) == length(w) || throw(ArgumentError("data and weight vectors must be the same size," *
708 "got $(length(v)) and $(length(w))"))
709 for x in w.values
710 isnan(x) && throw(ArgumentError("weight vector cannot contain NaN entries"))
711 x < 0 && throw(ArgumentError("weight vector cannot contain negative entries"))
712 end
713
714 isa(w, FrequencyWeights) && !(eltype(w) <: Integer) && any(!isinteger, w) &&
715 throw(ArgumentError("The values of the vector of `FrequencyWeights` must be numerically" *
716 "equal to integers. Use `ProbabilityWeights` or `AnalyticWeights` instead."))
717
718 # remove zeros weights and sort
719 wsum = sum(w)
720 nz = .!iszero.(w)
721 vw = sort!(collect(zip(view(v, nz), view(w, nz))))
722 N = length(vw)
723
724 # prepare percentiles
725 ppermute = sortperm(p)
726 p = p[ppermute]
727
728 # prepare out vector
729 out = Vector{typeof(zero(V)/1)}(undef, length(p))
730 fill!(out, vw[end][1])
731
732 @inbounds for x in v
733 isnan(x) && return fill!(out, x)
734 end
735
736 # loop on quantiles
737 Sk, Skold = zero(W), zero(W)
738 vk, vkold = zero(V), zero(V)
739 k = 0
740
741 w1 = vw[1][2]
742 for i in 1:length(p)
743 if isa(w, FrequencyWeights)
744 h = p[i] * (wsum - 1) + 1
745 else
746 h = p[i] * (wsum - w1) + w1
747 end
748 while Sk <= h
749 k += 1
750 if k > N
751 # out was initialized with maximum v
752 return out
753 end
754 Skold, vkold = Sk, vk
755 vk, wk = vw[k]
756 Sk += wk
757 end
758 if isa(w, FrequencyWeights)
759 out[ppermute[i]] = vkold + min(h - Skold, 1) * (vk - vkold)
760 else
761 out[ppermute[i]] = vkold + (h - Skold) / (Sk - Skold) * (vk - vkold)
762 end
763 end
764 return out
765 end
766
767 function quantile(v::RealVector, w::UnitWeights, p::RealVector)
768 length(v) != length(w) && throw(DimensionMismatch("Inconsistent array dimension."))
769 return quantile(v, p)
770 end
771
772 quantile(v::RealVector, w::AbstractWeights{<:Real}, p::Number) = quantile(v, w, [p])[1]
