41 """
42 `f_contiguous(T::Type, sz::NTuple{N,Int}, st::NTuple{N,Int})::Bool`
43
44 Whether an array is in column-major (Fortran, Julia) order, and
45 has elements stored contiguously. Any array that is f_contiguous will have
46 identical memory layout to a Julia `Array` of the same size.
47
48 `sz` should be the dimensions of the array in number of elements (i.e. what
49 `size(a)` would return)
50 `st` should be the stride(s) *in bytes* between elements in each dimension
51 """
52 function f_contiguous(::Type{T}, sz::NTuple{N,Int}, st::NTuple{N,Int}) where {T,N}
53 N == 0 && return true # 0-dimensional arrays have 1 element, always contiguous
54 if st[1] != sizeof(T)
55 # not contiguous
56 return false
57 end
58 if prod(sz) == 1 || length(sz) == 1
59 # 0 or 1-dim arrays (with correct stride) should default to f-contiguous
60 # in julia
61 return true
62 end
63 for j = 2:N
64 # check stride[cur_dim] == stride[prev_dim]*sz[prev_dim] for all dims>1,
65 # implying contiguous column-major storage (n.b. st[1] == sizeof(T) here)
66 if st[j] != st[j-1] * sz[j-1]
67 return false
68 end
69 end
70 return true
71 end

108 if !aligned(info)

116 end
117 return new{T,N}(o, info, tuple(info.sz...), div.(info.st, sizeof(T)),
118 f_contiguous(info), c_contiguous(info),

295 """
296 ```
297 NoCopyArray(o::PyObject)
298 ```
299 Convert a Python array-like object, to a Julia `Array` or `PermutedDimsArray`
300 without making a copy of the data.
301
302 If the data is stored in row-major format
303 (the default in Python/NumPy), then the returned array `nca` will be a
304 `PermutedDimsArray` such that the arrays are indexed the same way in Julia and
305 Python. i.e. `nca[idxs...] == o[idxs...]`
306
307 If the data is stored in column-major format then a regular Julia `Array` will
308 be returned.
309
310 Warning: This function is only lightly tested, and should be considered
311 experimental - it may cause segmentation faults on conversion or subsequent
312 array access, or be subtly broken in other ways. Only dense/contiguous, native
313 endian arrays that support the Python Buffer protocol are likely be converted
314 correctly.
315 """
316 function NoCopyArray(o::PyObject)
317 # n.b. the pydecref(::PyBuffer) finalizer handles releasing the PyBuffer
318 pybuf = PyBuffer(o, PyBUF_ND_STRIDED)
319 T, native_byteorder = array_format(pybuf)
320 !native_byteorder && throw(ArgumentError(
321 "Only native endian format supported, format string: '$(get_format_str(pybuf))'"))
322 T == Nothing && throw(ArgumentError(
323 "Array datatype '$(get_format_str(pybuf))' not supported"))
324 # TODO more checks on strides etc
325 sz = size(pybuf)
326 arr = unsafe_wrap(Array, convert(Ptr{T}, pybuf.buf.buf), sz, own=false)
327
328 !f_contiguous(T, sz, strides(pybuf)) &&
329 (arr = PermutedDimsArray(reshape(arr, reverse(sz)), (pybuf.buf.ndim:-1:1)))
330 return arr
331 end
