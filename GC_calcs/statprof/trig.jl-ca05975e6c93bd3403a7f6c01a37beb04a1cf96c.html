236 @inline function tan_kernel(y::DoubleFloat64, k)
237 # kernel tan function on ~[-pi/4, pi/4] (except on -0)
238 # Input y is assumed to be bounded by ~pi/4 in magnitude.
239 # Input k indicates whether tan (if k = 1) or -1/tan (if k = -1) is returned.
240
241 # Algorithm
242 # 1. Since tan(-y) = -tan(y), we need only to consider positive y.
243 # 2. Callers must return tan(-0) = -0 without calling here since our
244 # odd polynomial is not evaluated in a way that preserves -0.
245 # Callers may do the optimization tan(y) ~ y for tiny y.
246 # 3. tan(y) is approximated by a odd polynomial of degree 27 on
247 # [0,0.67434]
248 # 3 27
249 # tan(y) ~ y + T1*y + ... + T13*y ≡ P(y)
250 # where
251 #
252 # |tan(y) 2 4 26 | -59.2
253 # (tan(y)-P(y))/y = |----- - (1+T1*y +T2*y +.... +T13*y )| <= 2
254 # | y |
255 #
256 # Note: tan(y+z) = tan(y) + tan'(y)*z
257 # ~ tan(y) + (1+y*y)*z
258 # Therefore, for better accuracy in computing tan(y+z), let
259 # 3 2 2 2 2
260 # r = y *(T2+y *(T3+y *(...+y *(T12+y *T13))))
261 # then
262 # 3 2
263 # tan(y+z) = y + (T1*y + (y *(r+z)+z))
264 #
265 # 4. For y in [0.67434,pi/4], let z = pi/4 - y, then
266 # tan(y) = tan(pi/4-z) = (1-tan(z))/(1+tan(z))
267 # = 1 - 2*(tan(z) - (tan(z)^2)/(1+tan(z)))
268
269 yhi = y.hi
270 ylo = y.lo
271
272 if abs(yhi) >= 0.6744
273 if yhi < 0.0
274 yhi = -yhi
275 ylo = -ylo
276 end
277 # Then, accurately reduce y as "pio4hi"-yhi+"pio4lo"-ylo
278 yhi = (pi/4 - yhi) + (3.06161699786838301793e-17 - ylo)
279 # yhi is guaranteed to be exact, so ylo is identically zero
280 ylo = 0.0
281 end
282 y² = yhi * yhi
283 y⁴ = y² * y²
284
285 # Break P(y)-T1*y³ = y^5*(T[2]+y^2*T[3]+...) into y⁵*r + y⁵*v where
286 # r = T[2]+y^4*T[4]+...+y^20*T[12])
287 # v = (y^2*(T[3]+y^4*T[5]+...+y^22*[T13]))
288 r = @horner(y⁴,
289 1.33333333333201242699e-01, # T2
290 2.18694882948595424599e-02, # T4
291 3.59207910759131235356e-03, # T6
292 5.88041240820264096874e-04, # T8
293 7.81794442939557092300e-05, # T10
294 -1.85586374855275456654e-05) # T12

296 5.39682539762260521377e-02, # T3
297 8.86323982359930005737e-03, # T5
298 1.45620945432529025516e-03, # T7
299 2.46463134818469906812e-04, # T9
300 7.14072491382608190305e-05, # T11
301 2.59073051863633712884e-05) # T13
302 # Precompute y³
303 y³ = y² * yhi
304 # Calculate P(y)-y-T1*y³ = y⁵*r + y⁵*v = y²(y³*(r+v))
305 r = ylo + y² * (y³ * (r + v) + ylo)
306 # Calculate P(y)-y = r+T1*y³
307 r += 3.33333333333334091986e-01*y³
308 # Calculate w = r+y = P(y)
309 Px = yhi + r
310 if abs(y.hi) >= 0.6744
311 # If the original y was above the threshold, then we calculate
312 # tan(y) = 1 - 2*(tan(y) - (tan(y)^2)/(1+tan(y)))
313 # ≈ 1 - 2*(P(z) - (P(z)^2)/(1+P(z)))
314 # where z = y-π/4.
315 return (signbit(y.hi) ? -1.0 : 1.0)*(k - 2*(yhi-(Px^2/(k+Px)-r)))
316 end
317 if k == 1
318 # Else, we simply return w = P(y) if k == 1 (integer multiple from argument
319 # reduction was even)...
320 return Px
321 else
322 # ...or tan(y) ≈ -1.0/(y+r) if !(k == 1) (integer multiple from argument
323 # reduction was odd). If 2ulp error is allowed, simply return the frac-
324 # tion directly. Instead, we calculate it accurately.

334 end

864
865 Simultaneously compute [`sinpi(x)`](@ref) and [`cospi(x)`](@ref) (the sine and cosine of `π*x`,

869 This function requires Julia 1.6 or later.

875 isnan(x) && return x, x

880 s = maxintfloat(T)
881 ax >= s && return (copysign(zero(T), x), one(T)) # even integer-valued

897 else
898 y_si = mulpi_ext(copysign(one(T), rx) - rx)
899 y_co = mulpi_ext(one(T) - arx)
900 return sin_kernel(y_si), -cos_kernel(y_co)
901 end

965 else
966 Complex(F(NaN), F(NaN))
967 end
968 else

1011 function sincospi(z::Complex{T}) where T
1012 F = float(T)
1013 zr, zi = reim(z)
1014 if isinteger(zr)
1015 # zr = ...,-2,-1,0,1,2,...
1016 # sin(pi*zr) == ±0
1017 # cos(pi*zr) == ±1
1018 # cosh(pi*zi) > 0
1019 s = copysign(zero(F),zr)
1020 c_pos = isa(zr,Integer) ? iseven(zr) : isinteger(zr/2)

1023 (
1024 Complex(s, c_pos ? sh : -sh),

1026 )
1027 elseif isinteger(2*zr)
1028 # zr = ...,-1.5,-0.5,0.5,1.5,2.5,...
1029 # sin(pi*zr) == ±1

1031 # sign(sinh(pi*zi)) == sign(zi)
1032 s_pos = isinteger((2*zr-1)/4)
1033 pizi = pi*zi
1034 sh, ch = sinh(pizi), cosh(pizi)
1035 (
1036 Complex(s_pos ? ch : -ch, isnan(zi) ? zero(F) : copysign(zero(F),zi)),
1037 Complex(zero(F), s_pos ? -sh : sh),
1038 )
1039 elseif !isfinite(zr)
1040 if zi == 0
1041 Complex(F(NaN), F(zi)), Complex(F(NaN), isnan(zr) ? zero(F) : -flipsign(F(zi),zr))
1042 elseif isinf(zi)
1043 Complex(F(NaN), F(zi)), Complex(F(Inf), F(NaN))
1044 else
1045 Complex(F(NaN), F(NaN)), Complex(F(NaN), F(NaN))
1046 end
1047 else
1048 pizi = pi*zi
1049 sipi, copi = sincospi(zr)
1050 sihpi, cohpi = sinh(pizi), cosh(pizi)
1051 (
1052 Complex(sipi*cohpi, copi*sihpi),
1053 Complex(copi*cohpi, -sipi*sihpi),
1054 )

1091 """
1092 cosc(x)
1093
1094 Compute ``\\cos(\\pi x) / x - \\sin(\\pi x) / (\\pi x^2)`` if ``x \\neq 0``, and ``0`` if
1095 ``x = 0``. This is the derivative of `sinc(x)`.
1096 """

1098 function _cosc(x::Number)
1099 # naive cosc formula is susceptible to catastrophic
1100 # cancellation error near x=0, so we use the Taylor series
1101 # for small enough |x|.
1102 if fastabs(x) < 0.5
1103 # generic Taylor series: π ∑ (-1)^n (πx)^{2n-1}/a(n) where
1104 # a(n) = (1+2n)*(2n-1)! (= OEIS A174549)
1105 s = (term = -(π*x))/3
1106 π²x² = term^2
1107 ε = eps(fastabs(term)) # error threshold to stop sum
1108 n = 1
1109 while true
1110 n += 1
1111 term *= π²x²/((1-2n)*(2n-2))
1112 s += (δs = term/(1+2n))
1113 fastabs(δs) ≤ ε && break
1114 end
1115 return π*s
1116 else
1117 return isinf_real(x) ? zero(x) : ((pi*x)*cospi(x)-sinpi(x))/((pi*x)*x)
1118 end
1119 end

1123 fastabs(x) < 0.14 ? x*evalpoly(x^2, (-3.289868133696453, 3.2469697011334144, -1.1445109447325053, 0.2091827825412384, -0.023460810354558236, 0.001781145516372852)) :
1124 isinf_real(x) ? zero(x) : ((pi*x)*cospi(x)-sinpi(x))/((pi*x)*x)
