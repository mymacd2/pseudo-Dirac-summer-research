24 function _digamma(z::ComplexOrReal{Float64})
25 # Based on eq. (12), without looking at the accompanying source
26 # code, of: K. S. Kölbig, "Programs for computing the logarithm of
27 # the gamma function, and the digamma function, for complex
28 # argument," Computer Phys. Commun. vol. 4, pp. 221–226 (1972).
29 x = real(z)
30 if x <= 0 # reflection formula
31 ψ = -π * cot(π*z)
32 z = 1 - z
33 x = real(z)
34 else
35 ψ = zero(z)
36 end
37 if x < 7
38 # shift using recurrence formula
39 n = 7 - floor(Int,x)
40 for ν = 1:n-1
41 ψ -= inv(z + ν)
42 end
43 ψ -= inv(z)
44 z += n
45 end
46 t = inv(z)
47 ψ += log(z) - 0.5*t
48 t *= t # 1/z^2
49 # the coefficients here are Float64(bernoulli[2:9] .// (2*(1:8)))
50 ψ -= t * @evalpoly(t,0.08333333333333333,-0.008333333333333333,0.003968253968253968,-0.004166666666666667,0.007575757575757576,-0.021092796092796094,0.08333333333333333,-0.4432598039215686)

92 # (-1)^m d^m/dz^m cot(z) = p_m(cot z), where p_m is a polynomial
93 # that satisfies the recurrence p_{m+1}(x) = p_m′(x) * (1 + x^2).
94 # Note that p_m(x) has only even powers of x if m is odd, and
95 # only odd powers of x if m is even. Therefore, we write p_m(x)
96 # as p_m(x) = q_m(x^2) m! for m odd and p_m(x) = x q_m(x^2) m! if m is even.
97 # Hence the polynomials q_m(y) satisfy the recurrence:
98 # m odd: q_{m+1}(y) = 2 q_m′(y) * (1+y) / (m+1)
99 # m even: q_{m+1}(y) = [q_m(y) + 2 y q_m′(y)] * (1+y) / (m+1)
100 # This function computes the coefficients of the polynomial q_m(y),
101 # returning an array of the coefficients of 1, y, y^2, ...,
102 function cotderiv_q(m::Int)
103 m < 0 && throw(DomainError(m, "`m` must be nonnegative."))
104 m == 0 && return [1.0]
105 m == 1 && return [1.0, 1.0]
106 q₋ = cotderiv_q(m-1)
107 d = length(q₋) - 1 # degree of q₋
108 if isodd(m-1)
109 q = Vector{Float64}(undef, length(q₋))
110 q[end] = d * q₋[end] * 2/m
111 for i = 1:length(q)-1
112 q[i] = ((i-1)*q₋[i] + i*q₋[i+1]) * 2/m
113 end
114 else # iseven(m-1)
115 q = Vector{Float64}(undef, length(q₋) + 1)
116 q[1] = q₋[1] / m
117 q[end] = (1 + 2d) * q₋[end] / m
118 for i = 2:length(q)-1
119 q[i] = ((1 + 2(i-1))*q₋[i] + (1 + 2(i-2))*q₋[i-1]) / m
120 end
121 end
122 return q
123 end

128 # Evaluate (-1)^m d^m/dz^m [π cot(πz)] / m!. If m is small, we
129 # use the explicit derivative formula (a polynomial in cot(πz));
130 # if m is large, we use the derivative of Euler's harmonic series:
131 # π cot(πz) = ∑ inv(z + n)
132 function cotderiv(m::Integer, z)
133 isinf(imag(z)) && return zero(z)
134 if m <= 0
135 m == 0 && return π * cot(π*z)
136 throw(DomainError(m, "`m` must be nonnegative."))
137 end
138 if m <= length(cotderiv_Q)
139 q = cotderiv_Q[m]
140 x = cot(π*z)
141 y = x*x
142 s = q[1] + q[2] * y
143 t = y
144 # evaluate q(y) using Horner (TODO: Knuth for complex z?)
145 for i = 3:length(q)
146 t *= y
147 s += q[i] * t
148 end
149 return π^(m+1) * (isodd(m) ? s : x*s)
150 else # m is large, series derivative should converge quickly
151 p = m+1
152 z -= round(real(z))
153 s = inv(z^p)
154 n = 1
155 sₒ = zero(s)
156 while s != sₒ
157 sₒ = s
158 a = (z+n)^p
159 b = (z-n)^p
160 s += (a + b) / (a * b)
161 n += 1
162 end
163 return s
164 end

167 # Helper macro for polygamma(m, z):
168 # Evaluate p[1]*c[1] + x*p[2]*c[2] + x^2*p[3]*c[3] + ...
169 # where c[1] = m + 1
170 # c[k] = c[k-1] * (2k+m-1)*(2k+m-2) / ((2k-1)*(2k-2)) = c[k-1] * d[k]
171 # i.e. d[k] = c[k]/c[k-1] = (2k+m-1)*(2k+m-2) / ((2k-1)*(2k-2))
172 # by a modified version of Horner's rule:
173 # c[1] * (p[1] + d[2]*x * (p[2] + d[3]*x * (p[3] + ...))).
174 # The entries of p must be literal constants and there must be > 1 of them.
175 macro pg_horner(x, m, p...)
176 k = length(p)
177 me = esc(m)
178 xe = esc(x)
179 ex = :(($me + $(2k-1)) * ($me + $(2k-2)) * $(p[end]/((2k-1)*(2k-2))))
180 for k = length(p)-1:-1:2
181 cdiv = 1 / ((2k-1)*(2k-2))
182 ex = :(($cdiv * ($me + $(2k-1)) * ($me + $(2k-2))) *
183 ($(p[k]) + $xe * $ex))
184 end
185 :(($me + 1) * ($(p[1]) + $xe * $ex))
186 end

202 # Generalized zeta function, which is related to polygamma
203 # (at least for integer m > 0 and real(z) > 0) by:
204 # polygamma(m, z) = (-1)^(m+1) * gamma(m+1) * zeta(m+1, z).
205 # Our algorithm for the polygamma is just the m-th derivative
206 # of our digamma approximation, and this derivative process yields
207 # a function of the form
208 # (-1)^(m) * gamma(m+1) * (something)
209 # So identifying the (something) with the -zeta function, we get
210 # the zeta function for free and might as well export it, especially
211 # since this is a common generalization of the Riemann zeta function
212 # (which Julia already exports). Note that this generalization
213 # is equivalent to Mathematica's Zeta[s,z], and is equivalent to the
214 # Hurwitz zeta function for real(z) > 0.
215
216 """
217 zeta(s, z)
218
219 Generalized zeta function defined by
220 ```math
221 \\zeta(s, z)=\\sum_{k=0}^\\infty \\frac{1}{((k+z)^2)^{s/2}},
222 ```
223 where any term with ``k+z=0`` is excluded. For ``\\Re z > 0``,
224 this definition is equivalent to the Hurwitz zeta function
225 ``\\sum_{k=0}^\\infty (k+z)^{-s}``.
226
227 The Riemann zeta function is recovered as ``\\zeta(s)=\\zeta(s,1)``.
228
229 External links: [Riemann zeta function](https://en.wikipedia.org/wiki/Riemann_zeta_function), [Hurwitz zeta function](https://en.wikipedia.org/wiki/Hurwitz_zeta_function)
230 """
231 zeta(s::Number, z::Number) = _zeta(map(float, promote(s, z))...)

233 function _zeta(s::T, z::T) where {T<:ComplexOrReal{Float64}}
234 (z == 1 || z == 0) && return zeta(s)

237 # handle NaN cases

239 return T <: Real ? NaN : NaN + NaN*im

242 x = real(z)

244 # annoying s = Inf case:
245 if !isfinite(s)
246 if real(s) == Inf

248 return zero(T) # distance to poles is > 1

250 x > 0 && isreal(z) && isreal(s) && return T(Inf)

252 throw(DomainError(s, "`s` must be finite.")) # nothing clever to return

255 m = s - 1
256 ζ = zero(T)

258 # Algorithm is just the m-th derivative of digamma formula above,
259 # with a modified cutoff of the final asymptotic expansion.
260
261 # Note: we multiply by -(-1)^m m! in polygamma below, so this factor is
262 # pulled out of all of our derivatives.
263
264 cutoff = 7 + real(m) + abs(imag(m)) # TODO: this cutoff is too conservative?
265 if x < cutoff
266 # shift using recurrence formula
267 xf = floor(x)
268 nx = Int(xf)
269 n = ceil(Int, cutoff - nx)
270 minus_s = -s
271 if nx < 0 # x < 0
272 # need to use (-z)^(-s) recurrence to be correct for real z < 0
273 # [the general form of the recurrence term is (z^2)^(-s/2)]
274 minus_z = -z
275 ζ += pow_oftype(ζ, minus_z, minus_s) # ν = 0 term
276 if xf != z
277 ζ += pow_oftype(ζ, z - nx, minus_s)
278 # real(z - nx) > 0, so use correct branch cut
279 # otherwise, if xf==z, then the definition skips this term
280 end
281 # do loop in different order, depending on the sign of s,
282 # so that we are looping from largest to smallest summands and
283 # can halt the loop early if possible; see issue #15946
284 # FIXME: still slow for small m, large Im(z)
285 if real(s) > 0
286 for ν in -nx-1:-1:1
287 ζₒ= ζ
288 ζ += pow_oftype(ζ, minus_z - ν, minus_s)
289 ζ == ζₒ && break # prevent long loop for large -x > 0
290 end
291 else
292 for ν in 1:-nx-1
293 ζₒ= ζ
294 ζ += pow_oftype(ζ, minus_z - ν, minus_s)
295 ζ == ζₒ && break # prevent long loop for large -x > 0
296 end
297 end
298 else # x ≥ 0 && z != 0
299 ζ += pow_oftype(ζ, z, minus_s)
300 end
301 # loop order depends on sign of s, as above
302 if real(s) > 0
303 for ν in max(1,1-nx):n-1
304 ζₒ= ζ
305 ζ += pow_oftype(ζ, z + ν, minus_s)
306 ζ == ζₒ && break # prevent long loop for large m
307 end
308 else
309 for ν in n-1:-1:max(1,1-nx)
310 ζₒ= ζ
311 ζ += pow_oftype(ζ, z + ν, minus_s)
312 ζ == ζₒ && break # prevent long loop for large m
313 end
314 end
315 z += n
316 end
317
318 t = inv(z)
319 w = isa(t, Real) ? conj(oftype(ζ, t))^m : oftype(ζ, t)^m
320 ζ += w * (inv(m) + 0.5*t)
321
322 t *= t # 1/z^2
323 ζ += w*t * @pg_horner(t,m,0.08333333333333333,-0.008333333333333333,0.003968253968253968,-0.004166666666666667,0.007575757575757576,-0.021092796092796094,0.08333333333333333,-0.4432598039215686,3.0539543302701198)
324
325 return ζ
326 end

340 function _polygamma(m::Integer, z::ComplexOrReal{Float64})
341 m == 0 && return digamma(z)
342 m == 1 && return trigamma(z)
343
344 # In principle, we could support non-integer m here, but the
345 # extension to complex m seems to be non-unique, the obvious
346 # extension (just plugging in a complex m below) does not seem to
347 # be the one used by Mathematica or Maple, and sources do not
348 # agree on what the "right" extension is (e.g. Mathematica & Maple
349 # disagree). So, at least for now, we require integer m.
350
351 # real(m) < 0 is valid, but I don't think our asymptotic expansion
352 # here works in this case. m < 0 polygamma is called a
353 # "negapolygamma" function in the literature, and there are
354 # multiple possible definitions arising from different integration
355 # constants. We throw a DomainError since the definition is unclear.
356 real(m) < 0 && throw(DomainError(m, "`real(m)` must be nonnegative, since the definition in this case is ambiguous."))
357
358 s = Float64(m+1)
359 # It is safe to convert any integer (including `BigInt`) to a float here
360 # as underflow occurs before precision issues.
361 if real(z) <= 0 # reflection formula
362 (zeta(s, 1-z) + signflip(m, cotderiv(m,z))) * (-gamma(s))
363 else
364 signflip(m, zeta(s,z) * (-gamma(s)))
365 end
366 end

375 function _invdigamma(y::Float64)
376 # Implementation of fixed point algorithm described in
377 # "Estimating a Dirichlet distribution" by Thomas P. Minka, 2000
378
379 # Closed form initial estimates
380 if y >= -2.22
381 x_old = exp(y) + 0.5
382 x_new = x_old
383 else
384 x_old = -1.0 / (y - digamma(1.0))
385 x_new = x_old
386 end
387
388 # Fixed point algorithm
389 delta = Inf
390 iteration = 0
391 while delta > 1e-12 && iteration < 25
392 iteration += 1
393 x_new = x_old - (digamma(x_old) - y) / trigamma(x_old)
394 delta = abs(x_new - x_old)
395 x_old = x_new
396 end
397
398 return x_new
399 end

414 function _zeta(s::ComplexOrReal{Float64})
415 # Riemann zeta function; algorithm is based on specializing the Hurwitz
416 # zeta function above for z==1.
417
418 # blows up to ±Inf, but get correct sign of imaginary zero
419 s == 1 && return NaN + zero(s) * imag(s)
420
421 if !isfinite(s) # annoying NaN and Inf cases
422 isnan(s) && return imag(s) == 0 ? s : s*s
423 if isfinite(imag(s))
424 real(s) > 0 && return 1.0 - zero(s)*imag(s)
425 imag(s) == 0 && return NaN + zero(s)
426 end
427 return NaN*zero(s) # NaN + NaN*im
428 elseif real(s) < 0.5
429 absim = abs(imag(s))
430 if abs(real(s)) + absim < 1e-3 # Taylor series for small |s|
431 return @evalpoly(s, -0.5,
432 -0.918938533204672741780329736405617639861,
433 -1.0031782279542924256050500133649802190,
434 -1.00078519447704240796017680222772921424,
435 -0.9998792995005711649578008136558752359121)
436 end
437 if absim > 12 # amplitude of sinpi(s/2) ≈ exp(imag(s)*π/2)
438 # avoid overflow/underflow (issue #128)
439 lg = loggamma(1 - s)
440 rehalf = real(s)*0.5
441 return zeta(1 - s) * exp(lg + absim*halfπ + s*log2π) * inv2π * Complex(
442 sinpi(rehalf), flipsign(cospi(rehalf), imag(s))
443 )
444 else
445 return zeta(1 - s) * gamma(1 - s) * sinpi(s*0.5) * twoπ^s * invπ
446 end
447 end
448
449 m = s - 1
450
451 # shift using recurrence formula:
452 # n is a semi-empirical cutoff for the Stirling series, based
453 # on the error term ~ (|m|/n)^18 / n^real(m)
454 # FIXME: slow for large imag(s) and small real(s)
455 n = ceil(Int,6 + 0.7*abs(imag(s-1))^inv(1 + real(m)*0.05))
456 ζ = one(s)
457 for ν = 2:n
458 ζₒ= ζ
459 ζ += inv(ν)^s
460 ζ == ζₒ && break # prevent long loop for large m
461 end
462 z = 1 + n
463 t = inv(z)
464 w = t^m
465 ζ += w * (inv(m) + 0.5*t)
466
467 t *= t # 1/z^2
468 ζ += w*t * @pg_horner(t,m,0.08333333333333333,-0.008333333333333333,0.003968253968253968,-0.004166666666666667,0.007575757575757576,-0.021092796092796094,0.08333333333333333,-0.4432598039215686,3.0539543302701198)

651 # Compute the logΓ(z) function using a combination of the asymptotic series,
652 # the Taylor series around z=1 and z=2, the reflection formula, and the shift formula.
653 # Many details of these techniques are discussed in D. E. G. Hare,
654 # "Computing the principal branch of log-Gamma," J. Algorithms 25, pp. 221-236 (1997),
655 # and similar techniques are used (in a somewhat different way) by the
656 # SciPy loggamma function. The key identities are also described
657 # at http://functions.wolfram.com/GammaBetaErf/LogGamma/
658 function _loggamma(z::Complex{Float64})
659 x, y = reim(z)
660 yabs = abs(y)
661 if !isfinite(x) || !isfinite(y) # Inf or NaN
662 if isinf(x) && isfinite(y)
663 return Complex(x, x > 0 ? (y == 0 ? y : copysign(Inf, y)) : copysign(Inf, -y))
664 elseif isfinite(x) && isinf(y)
665 return Complex(-Inf, y)
666 else
667 return Complex(NaN, NaN)
668 end
669 elseif x > 7 || yabs > 7 # use the Stirling asymptotic series for sufficiently large x or |y|
670 return loggamma_asymptotic(z)
671 elseif x < 0.1 # use reflection formula to transform to x > 0
672 if x == 0 && y == 0 # return Inf with the correct imaginary part for z == 0
673 return Complex(Inf, signbit(x) ? copysign(Float64(π), -y) : -y)
674 end
675 # the 2pi * floor(...) stuff is to choose the correct branch cut for log(sinpi(z))
676 return Complex(Float64(logπ), copysign(Float64(twoπ), y) * floor(0.5*x+0.25)) -
677 log(sinpi(z)) - loggamma(1-z)
678 elseif abs(x - 1) + yabs < 0.1
679 # taylor series at z=1
680 # ... coefficients are [-eulergamma; [(-1)^k * zeta(k)/k for k in 2:15]]
681 w = Complex(x - 1, y)
682 return w * @evalpoly(w, -5.7721566490153286060651188e-01,8.2246703342411321823620794e-01,
683 -4.0068563438653142846657956e-01,2.705808084277845478790009e-01,
684 -2.0738555102867398526627303e-01,1.6955717699740818995241986e-01,
685 -1.4404989676884611811997107e-01,1.2550966952474304242233559e-01,
686 -1.1133426586956469049087244e-01,1.000994575127818085337147e-01,
687 -9.0954017145829042232609344e-02,8.3353840546109004024886499e-02,
688 -7.6932516411352191472827157e-02,7.1432946295361336059232779e-02,
689 -6.6668705882420468032903454e-02)
690 elseif abs(x - 2) + yabs < 0.1
691 # taylor series at z=2
692 # ... coefficients are [1-eulergamma; [(-1)^k * (zeta(k)-1)/k for k in 2:12]]
693 w = Complex(x - 2, y)
694 return w * @evalpoly(w, 4.2278433509846713939348812e-01,3.2246703342411321823620794e-01,
695 -6.7352301053198095133246196e-02,2.0580808427784547879000897e-02,
696 -7.3855510286739852662729527e-03,2.8905103307415232857531201e-03,
697 -1.1927539117032609771139825e-03,5.0966952474304242233558822e-04,
698 -2.2315475845357937976132853e-04,9.9457512781808533714662972e-05,
699 -4.4926236738133141700224489e-05,2.0507212775670691553131246e-05)
700 end
701 # use recurrence relation loggamma(z) = loggamma(z+1) - log(z) to shift to x > 7 for asymptotic series
702 shiftprod = Complex(x,yabs)
703 x += 1
704 sb = false # == signbit(imag(shiftprod)) == signbit(yabs)
705 # To use log(product of shifts) rather than sum(logs of shifts),
706 # we need to keep track of the number of + to - sign flips in
707 # imag(shiftprod), as described in Hare (1997), proposition 2.2.
708 signflips = 0
709 while x <= 7
710 shiftprod *= Complex(x,yabs)
711 sb′ = signbit(imag(shiftprod))
712 signflips += sb′ & (sb′ != sb)
713 sb = sb′
714 x += 1
715 end
716 shift = log(shiftprod)
717 if signbit(y) # if y is negative, conjugate the shift
718 shift = Complex(real(shift), signflips*-6.2831853071795864769252842 - imag(shift))
719 else
720 shift = Complex(real(shift), imag(shift) + signflips*6.2831853071795864769252842)
721 end
722 return loggamma_asymptotic(Complex(x,y)) - shift
723 end

725 # asymptotic series for log(gamma(z)), valid for sufficiently large real(z) or |imag(z)|
726 function loggamma_asymptotic(z::Complex{Float64})
727 zinv = inv(z)
728 t = zinv*zinv
729 # coefficients are bernoulli[2:n+1] .// (2*(1:n).*(2*(1:n) - 1))
730 return (z-0.5)*log(z) - z + 9.1893853320467274178032927e-01 + # <-- log(2pi)/2
731 zinv*@evalpoly(t, 8.3333333333333333333333368e-02,-2.7777777777777777777777776e-03,
732 7.9365079365079365079365075e-04,-5.9523809523809523809523806e-04,
733 8.4175084175084175084175104e-04,-1.9175269175269175269175262e-03,
734 6.4102564102564102564102561e-03,-2.9550653594771241830065352e-02)
735 end

770 """
771 logabsbeta(x, y)
772
773 Compute the natural logarithm of the absolute value of the [`beta`](@ref) function, returning a tuple `(log(abs(beta(x,y))), sign(beta(x,y)))`
774
775 See also [`logbeta`](@ref).
776 """
777 function logabsbeta(a::T, b::T) where T<:Real
778 # ensure that a <= b
779 if a > b
780 return logabsbeta(b, a)
781 end
782
783 if a <= 0 && isinteger(a)
784 if a + b <= 0 && isinteger(b)
785 r = logbeta(1 - a - b, b)
786 # in julia ≥ 1.7, iseven doesn't require Int (julia#38976)
787 sgn = iseven(@static VERSION ≥ v"1.7" ? b : Int(b)) ? 1 : -1
788 return r, sgn
789 else
790 return -log(zero(a)), 1
791 end
792 end
793
794 # asymptotic expansion for large `b` and positive arguments
795 # FIXME! We probably lose precision for negative arguments. However, `loggammadiv`
796 # is based on the NSWC code which doesn't bother with negative arguments
797 if a > 0 && b > 8
798 return (loggammadiv(a, b) + loggamma(a)), 1
799 end
800
801 ya, sa = logabsgamma(a)
802 yb, sb = logabsgamma(b)
803 yab, sab = logabsgamma(a + b)
804 (ya + yb - yab), Int(sa*sb*sab)
805 end
806 logabsbeta(a::Real, b::Real) = logabsbeta(promote(a, b)...)

812 """
813 logabsbinomial(n, k)
814
815 Accurate natural logarithm of the absolute value of the [`binomial`](@ref)
816 coefficient `binomial(n, k)` for large `n` and `k` near `n/2`.
817
818 Returns a tuple `(log(abs(binomial(n,k))), sign(binomial(n,k)))`.
819 """
820 function logabsbinomial(n::T, k::T) where {T<:Integer}
821 S = float(T)
822 s = one(S)
823 if n < 0
824 # reflection formula
825 # binomial(n,k) = (-1)^k * binomial(-n+k-1,k)
826 n = -n + k - 1
827 s = isodd(k) ? -s : s
828 end
829 if k < 0 || k > n
830 # binomial(n,k) == 0
831 return (typemin(S), zero(S))
832 elseif k == 0 || k == n
833 # binomial(n,k) == ±1
834 return (zero(S), s)
835 end
836 if k > (n>>1)
837 k = n - k
838 end
839 if k == 1
840 # binomial(n,k) == ±n
841 return (log(abs(n)), s)
842 else
843 return (-log1p(n) - logabsbeta(n - k + one(T), k + one(T))[1], s)
844 end
845 end
846 logabsbinomial(n::Integer, k::Integer) = logabsbinomial(promote(n, k)...)
